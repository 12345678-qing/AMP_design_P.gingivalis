# ===================== Model Comparison: 5-Fold Cross-Validation =====================
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import warnings
warnings.filterwarnings('ignore')

# -------------------------- Set random seed for reproducibility --------------------------
def set_all_seeds(seed=42):
    np.random.seed(seed)
    import random
    random.seed(seed)
    import os
    os.environ['PYTHONHASHSEED'] = str(seed)

set_all_seeds(seed=42)

# -------------------------- Load and clean data --------------------------
data_path = "Pg_AMP_data.csv"
df = pd.read_csv(data_path, encoding='utf-8')

df['cLogP'] = df['cLogP'].astype(str).str.replace('?', '').str.replace('-', '').astype(float)
df = df[df['Class_gingivalis'].isin([0, 1])]
df_clean = df.dropna(subset=['cLogP','Net_charge','Length','Cationic_comp','Hydrophobic_comp'])

features = ['cLogP','Net_charge','Length','Cationic_comp','Hydrophobic_comp']
X = df_clean[features]
y = df_clean['Class_gingivalis']

# -------------------------- 5-Fold Stratified Cross-Validation --------------------------
from sklearn.model_selection import StratifiedKFold
skf = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)

# -------------------------- Model definition --------------------------
from sklearn.ensemble import RandomForestClassifier
from sklearn.linear_model import LogisticRegression
from xgboost import XGBClassifier

models = {
    "Random Forest": RandomForestClassifier(
        n_estimators=300,
        max_depth=8,
        min_samples_split=2,
        min_samples_leaf=1,
        max_features="sqrt",
        criterion="entropy",
        class_weight="balanced",
        random_state=42,
        n_jobs=-1
    ),
    "Logistic Regression": LogisticRegression(
        class_weight="balanced", max_iter=2000, random_state=42
    ),
    "XGBoost": XGBClassifier(
        n_estimators=100,
        scale_pos_weight=len(y[y==0])/len(y[y==1]),
        random_state=42,
        eval_metric="logloss",
        use_label_encoder=False
    )
}

# -------------------------- Run 5-fold cross-validation --------------------------
from sklearn.metrics import average_precision_score, accuracy_score, recall_score, precision_score, f1_score

results = []
all_fold_data = []

for model_name, model in models.items():
    ap_list = []
    acc_list = []
    recall_list = []
    precision_list = []
    f1_list = []

    for train_idx, val_idx in skf.split(X, y):
        X_train, X_val = X.iloc[train_idx], X.iloc[val_idx]
        y_train, y_val = y.iloc[train_idx], y.iloc[val_idx]

        model.fit(X_train, y_train)
        y_pred = model.predict(X_val)
        y_prob = model.predict_proba(X_val)[:, 1]

        ap = average_precision_score(y_val, y_prob)
        acc = accuracy_score(y_val, y_pred)
        recall = recall_score(y_val, y_pred, zero_division=0)
        precision = precision_score(y_val, y_pred, zero_division=0)
        f1 = f1_score(y_val, y_pred, zero_division=0)

        ap_list.append(ap)
        acc_list.append(acc)
        recall_list.append(recall)
        precision_list.append(precision)
        f1_list.append(f1)

    for ap_val in ap_list:
        all_fold_data.append({"model": model_name, "ap": ap_val})

    results.append({
        "Model": model_name,
        "AP": f"{np.mean(ap_list):.4f}±{np.std(ap_list):.4f}",
        "Accuracy": f"{np.mean(acc_list):.4f}±{np.std(acc_list):.4f}",
        "Recall": f"{np.mean(recall_list):.4f}±{np.std(recall_list):.4f}",
        "Precision": f"{np.mean(precision_list):.4f}±{np.std(precision_list):.4f}",
        "F1": f"{np.mean(f1_list):.4f}±{np.std(f1_list):.4f}"
    })

# -------------------------- Save results --------------------------
table_df = pd.DataFrame(results)
print("\n" + "="*60)
print("5-Fold Cross-Validation Results")
print("="*60)
print(table_df.to_string(index=False))

table_df.to_csv("5fold_cv_results.csv", index=False, encoding='utf-8-sig')
fold_df = pd.DataFrame(all_fold_data)

# -------------------------- Plot boxplot --------------------------
plt.rcParams['font.family'] = 'serif'
plt.rcParams['font.serif'] = ['Times New Roman', 'DejaVu Serif']
plt.rcParams['font.size'] = 13

fig, ax = plt.subplots(figsize=(10, 7), dpi=300)

model_names = ["Random Forest", "Logistic Regression", "XGBoost"]
data = [fold_df[fold_df["model"] == m]["ap"].values for m in model_names]
colors = ['#e66150', '#67b8d8', '#38b09d']

bp = ax.boxplot(
    data,
    labels=model_names,
    patch_artist=True,
    medianprops={'color': 'white', 'linewidth': 2},
    whiskerprops={'color': 'black', 'linewidth': 1.2},
    capprops={'color': 'black', 'linewidth': 1.2},
    boxprops={'linewidth': 1.2},
    flierprops={'marker': 'o', 'markerfacecolor': 'black', 'markersize': 6}
)

for patch, color in zip(bp['boxes'], colors):
    patch.set_facecolor(color)
    patch.set_alpha(0.8)

for i, m in enumerate(model_names):
    mean_val = fold_df[fold_df["model"] == m]["ap"].mean()
    ax.scatter(i+1, mean_val, color='red', s=150, zorder=5, label='Mean AP' if i==0 else "")
    ax.text(i+1, mean_val + 0.008, f"{mean_val:.4f}", ha='center', va='bottom', fontsize=12, fontweight='bold')

ax.set_ylim(0.6, 1.0)
ax.set_ylabel('Average Precision (AP)', fontsize=15, fontweight='bold')
ax.set_xlabel('Machine Learning Models', fontsize=15, fontweight='bold')
ax.set_title('5-Fold Cross-Validation AP Comparison', fontsize=16, fontweight='bold', pad=20)
ax.grid(axis='y', linestyle='-', alpha=0.3)
ax.legend(loc='upper right', fontsize=13)

plt.tight_layout()
plt.savefig("figure_5fold_ap_boxplot.png", dpi=300, bbox_inches='tight', facecolor='white')
plt.show()

print("\nDone! All outputs saved successfully.")
print("→ 5fold_cv_results.csv")
print("→ figure_5fold_ap_boxplot.png")
