import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import shap
import warnings
warnings.filterwarnings('ignore')

def set_all_seeds(seed=42):
    np.random.seed(seed)
    import random
    random.seed(seed)
    import os
    os.environ['PYTHONHASHSEED'] = str(seed)
    from sklearn import set_config
    set_config(enable_metadata_routing=False)

set_all_seeds(seed=42)

data_path = "Pg_AMP_data.csv"
df = pd.read_csv(data_path, encoding='utf-8')

features = ['cLogP', 'Net_charge', 'Length', 'Cationic_comp', 'Hydrophobic_comp', 'Has_neutral']
X = df[features].copy()
y = df['Class_gingivalis'].copy()

from sklearn.ensemble import RandomForestClassifier
best_rf = RandomForestClassifier(
    n_estimators=300,
    max_depth=8,
    criterion="entropy",
    max_features="sqrt",
    class_weight="balanced",
    random_state=42,
    n_jobs=-1
)
best_rf.fit(X, y)

explainer = shap.TreeExplainer(best_rf)
shap_values = explainer.shap_values(X)

shap_pos = shap_values[:, :, 1]

mean_abs_shap = np.abs(shap_pos).mean(axis=0)

assert mean_abs_shap.ndim == 1, "Error: mean_abs_shap must be a 1-dimensional array!"
assert len(mean_abs_shap) == len(features), f"Error: length mismatch ({len(mean_abs_shap)} vs {len(features)})"
shap_df = pd.DataFrame({
    'Feature': features,
    'Mean_Abs_SHAP': mean_abs_shap
}).sort_values(by='Mean_Abs_SHAP', ascending=True)  

print(shap_df)

plt.rcParams.update({
    'font.family': 'serif',
    'font.serif': ['Times New Roman'],
    'font.size': 12,
    'axes.linewidth': 1.5,
    'axes.spines.top': False,
    'axes.spines.right': False,
    'grid.alpha': 0.3,
    'figure.dpi': 400,
    'savefig.dpi': 400
})

fig, ax = plt.subplots(figsize=(8, 5))
bar_color = '#67b8d8'  
ax.barh(
    y=shap_df['Feature'],
    width=shap_df['Mean_Abs_SHAP'],
    color=bar_color,
    edgecolor='black',
    linewidth=1.0
)

ax.set_xlabel('Mean Absolute SHAP Value', fontsize=13, fontweight='bold')
ax.set_ylabel('Feature', fontsize=13, fontweight='bold')
ax.set_title('Feature Importance for P. gingivalis AMPs (RF + SHAP)', fontsize=14, fontweight='bold', pad=15)
ax.grid(axis='x', linestyle='--')

plt.tight_layout()

plt.savefig(
    "SHAP_MeanAbs_Pg_AMP_RF_Blue.png",
    bbox_inches='tight',
    facecolor='white'
)
plt.show()

print("\n" + "="*60)
print("Mean Absolute SHAP Value ）")
print("="*60)
print(shap_df.sort_values(by='Mean_Abs_SHAP', ascending=False).to_string(index=False))
